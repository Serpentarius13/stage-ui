## Install

```
yarn add @stage-ui/core
```

## Getting started

```
import React from 'react'
import { Viewport, Header } from '@stage-ui/core'

export default () => {
    return (
        <Viewport>
            <Header>Hello world!</Header>
        </Viewport>
    )
}
```
